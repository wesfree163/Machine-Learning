# Define transformations to apply to the images
transform = transforms.Compose([
    transforms.Resize(256),
    transforms.CenterCrop(224),
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])
])

# Create the dataset
dataset = torchvision.datasets.ImageFolder(root='C:\\Users\\Wes\\Downloads\\DEXP-IMG', transform=transform)

# Create a DataLoader for batching and shuffling

dataloader = torchvision.utils.save_image.DataLoader(dataset, batch_size=17, shuffle=True, num_workers=4)
